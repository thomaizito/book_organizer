@app.route('/', methods=['GET', "POST"])

        turma = request.form['turma']

        horario.esc(turma)
        horario.today_day()

        for i in horario:
            if i.lower() in items:
                livros_do_dia[i.lower()] = items[i.lower()]
                

        livros.apd(livros_do_dia)
